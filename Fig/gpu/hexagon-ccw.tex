\documentclass[tikz,border=5pt]{standalone}
\usepackage{tikz}
\usetikzlibrary{calc,decorations.markings}

\begin{document}

% Classic latex arrowheads (compatible everywhere)
\tikzset{
  >={latex},
  edgeccw/.style={
    postaction={
      decorate,
      decoration={
        markings,
        mark=at position 0.55 with {
          \draw[->] (0,0) arc[start angle=0, end angle=180, radius=8pt];
          \node at (0,6pt) {$\theta$};
        }
      }
    }
  }
}

\begin{tikzpicture}[scale=2, line width=0.8pt]

  % Regular hexagon vertices
  \coordinate (P0) at (1,0);
  \coordinate (P1) at (2,0.866);
  \coordinate (P2) at (2,2.598);
  \coordinate (P3) at (1,3.464);
  \coordinate (P4) at (0,2.598);
  \coordinate (P5) at (0,0.866);

  % CCW edges with curved Î¸ arcs
  \draw[->,edgeccw] (P0) -- (P1);
  \draw[->,edgeccw] (P1) -- (P2);
  \draw[->,edgeccw] (P2) -- (P3);
  \draw[->,edgeccw] (P3) -- (P4);
  \draw[->,edgeccw] (P4) -- (P5);
  \draw[->,edgeccw] (P5) -- (P0);

  % Vertex markers
  \fill (P0) circle (0.03) node[right] {$P_0$};
  \fill (P1) circle (0.03) node[right] {$P_1$};
  \fill (P2) circle (0.03) node[right] {$P_2$};
  \fill (P3) circle (0.03) node[left] {$P_3$};
  \fill (P4) circle (0.03) node[left] {$P_4$};
  \fill (P5) circle (0.03) node[left] {$P_5$};

\end{tikzpicture}
\end{document}
