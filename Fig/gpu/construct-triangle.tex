\documentclass[tikz,border=5pt]{standalone}
\usepackage{tikz}
\usepackage{tikz-3dplot}

\begin{document}

% Set viewing angle
\tdplotsetmaincoords{70}{120}

\begin{tikzpicture}[tdplot_main_coords, scale=2]

  % Axes
  \draw[->] (0,0,0) -- (3,0,0) node[anchor=north east] {$x$};
  \draw[->] (0,0,0) -- (0,3,0) node[anchor=north west] {$y$};
  \draw[->] (0,0,0) -- (0,0,3) node[anchor=south] {$z$};

  % Triangle vertices
  \coordinate (A) at (1,1,1);
  \coordinate (B) at (2,2,1);
  \coordinate (C) at (1,2,1);

  % Eye
  \coordinate (Pout) at (1,1.5,2);
  \coordinate (Pin) at (1,1.5,0);

  % Triangle edges (CCW)
  %  1. `->` is shorthand for “use the default arrow tip”
  %  2. `-latex` explicitly selects the `latex` arrow tip
  \draw[thick, blue, -latex] (A) -- (B);
  \draw[thick, blue, -latex] (B) -- (C);
  \draw[thick, blue, -latex] (C) -- (A);

  % Vertex markers
  \fill (A) circle (0.04) node[below left] {$v_0 (1,1,1)$};
  \fill (B) circle (0.04) node[below] {$v_1 (2,2,1)$};
  \fill (C) circle (0.04) node[below right] {$v_2 (1,2,1)$};

  % Eye
  \fill[blue]   (Pout) circle (0.04)
    node[above right] {$p_0$};

  \fill[blue] (Pin) circle (0.04)
    node[below left] {$p_1$};

\end{tikzpicture}

\end{document}
