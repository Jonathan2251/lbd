\documentclass[tikz,border=5pt]{standalone}
\usepackage{tikz}
\usetikzlibrary{calc,decorations.markings}

\begin{document}

% Classic latex arrowheads (compatible everywhere)
\tikzset{
  >={latex},
  edgecw/.style={
    postaction={
      decorate,
      decoration={
        markings,
        mark=at position 0.55 with {
          \draw[<-] (0,0) arc[start angle=180, end angle=360, radius=8pt];
          \node at (10pt,-9pt) {$\theta$};
        }
      }
    }
  }
}

\begin{tikzpicture}[scale=2, line width=0.8pt]

  % Vertices
  \coordinate (P0) at (0,0);
  \coordinate (P1) at (1,1.732);
  \coordinate (P2) at (2,0);

  % Triangle edges with CCW curved Î¸ on the edges
  \draw[->,edgecw] (P0) -- (P1);
  \draw[->,edgecw] (P1) -- (P2);
  \draw[->,edgecw] (P2) -- (P0);

  % Vertex markers
  \fill (P0) circle (0.03) node[below left] {$P_0$};
  \fill (P1) circle (0.03) node[below right] {$P_1$};
  \fill (P2) circle (0.03) node[above] {$P_2$};

\end{tikzpicture}
\end{document}
